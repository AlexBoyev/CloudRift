pipeline {
    agent any
    
    triggers {
        githubPush()
    }

    stages {
        stage('Setup') {
            steps {
                script {
                    try {
                        checkout scm
                        sh '''
                            echo "Workspace contents:"
                            ls -la
                            
                            echo "Python version:"
                            python3 --version
                            
                            # Create and activate venv
                            python3 -m venv .venv
                            source .venv/bin/activate
                        '''
                    } catch (Exception e) {
                        error "Setup failed: ${e.message}"
                    }
                }
            }
        }

        stage('Run Tests') {
            steps {
                script {
                    try {
                        sh '''
                            source .venv/bin/activate
                            
                            # Go to test directory and run tests
                            cd Testing/Tree
                            python3 tree-testing.py
                        '''
                    } catch (Exception e) {
                        error "Tests failed: ${e.message}"
                    }
                }
            }
        }
    }

    post {
        always {
            cleanWs()
        }
    }
}
